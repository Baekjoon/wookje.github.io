[3683번: 고양이와 개](https://www.acmicpc.net/problem/3683)

## 풀이

이분 매칭으로 풀면 된다.  

## 코드

```cpp
#include <bits/stdc++.h>
using namespace std;

vector<int> A[501];
string v1[501], v2[501];
int visited[501];
bool chk[501];

void clean() {
	for (int i = 0; i < 501; ++i) {
		A[i].clear();
		v1[i].clear();
		v2[i].clear();
		visited[i] = -1;
	}
}

int BM(int a) {
	for (auto b : A[a]) {
		if (chk[b]) continue;
		chk[b] = true;
		if (visited[b] == -1 || BM(visited[b])) {
			visited[b] = a;
			return true;
		}
	}
	return false;
}

int main() {
	int tc;
	scanf("%d", &tc);
	while (tc--) {
		clean();
		int c, d, v;
		scanf("%d %d %d", &c, &d, &v);
		for (int i = 0; i < v; ++i)
			cin >> v1[i] >> v2[i];

		for (int i = 0; i < v - 1; ++i) {
			for (int j = i + 1; j < v; ++j) {
				if (v1[i].compare(v2[j]) == 0 || v1[j].compare(v2[i]) == 0) {
					if (v1[i][0] == 'C') A[i].push_back(j);
					else A[j].push_back(i);
				}
			}
		}

		int sum = 0;
		for (int i = 0; i < v; ++i) {
			memset(chk, 0, sizeof(chk));
			if (BM(i)) ++sum;
		}
		printf("%d\n", v - sum);
	}

	return 0;
}
```
