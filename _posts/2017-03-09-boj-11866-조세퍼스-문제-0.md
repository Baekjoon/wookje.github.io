[11866번: 조세퍼스 문제 0](https://www.acmicpc.net/problem/11866)

## 풀이

단순 구현  

## 코드

```cpp
#include <bits/stdc++.h>
using namespace std;

int n, m, a[1010];
int cnt, pos = -1;
vector<int> ans;

int main() {
	cin >> n >> m;

	for (int i = 0; i < n; i++) a[i] = i + 1;
	while (1) {
		if (ans.size() == n) break;
		pos = (pos + 1) % n;
		if (!a[pos]) continue;
		cnt++;
		if (cnt == m) {
			ans.push_back(a[pos]);
			cnt = a[pos] = 0;
		}
	}
	printf("<%d", ans[0]);
	for (int i = 1; i < ans.size(); i++) printf(", %d", ans[i]);
	printf(">");

	return 0;
}
```
