## 코드

```cpp
#include <stdio.h>
#include <algorithm>
using namespace std;

int n;
pair<int, int> a[400004];

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n * 2; i += 2) {
		int s, t;
		scanf("%d %d", &s, &t);
		a[i] = { s,1 };
		a[i + 1] = { t,-1 };
	}

	sort(a, a + n * 2);

	int ans = 0, tmp = 0;
	for (int i = 0; i < n * 2; i++) {
		ans = max(ans, tmp += a[i].second);
	}

	printf("%d", ans);

	return 0;
}
```
